<h2><a name="p-9516681-h-1-asr-cat_with_wry_smile-1" class="anchor" href="https://linux.do#p-9516681-h-1-asr-cat_with_wry_smile-1"></a>1. 支持ASR本地模型 <img src="https://linux.do/images/emoji/twemoji/cat_with_wry_smile.png?v=14" title=":cat_with_wry_smile:" class="emoji" alt=":cat_with_wry_smile:" loading="lazy" width="20" height="20"></h2>
<p>看见不少佬友对ASR隐私问题存在顾虑,调研一波之后,目前已支持sensevoice small作为本地模型,共有两个版本:</p>
<ul>
<li>int8版本,模型体积226mb,运行时软件内存占用共约700MB,实测8gen3机型预热模型需要4s左右,加载完成后启用NNAPI以及4线程推理,识别单句话的延迟在70-80ms左右,非常迅速</li>
<li>无量化版本,模型体积886mb,运行时软件内存占用共约2G,实测8gen3机型预热模型需要8s左右,加载完成后识别单句话的延迟在200ms左右,延迟依然远低于在线模型</li>
<li>后续可能还会加入TeleSpeechASR模型,在方言识别上更具优势.</li>
<li>PS:模型解压和验证所需时间较长,请耐心等待,防止文件损坏导致应用闪退</li>
</ul>
<p>具体使用上个人感觉sensevoice small full完全能满足一般中英文需求,这两个模在硅基流动渠道都有免费的api,可以先体验一下效果</p>

<li>支持了硅基流动的qwen3-omni系列</li>

<h3><a name="p-9516662-h-2" class="anchor" href="https://linux.do#p-9516662-h-2"></a>脚本介绍：</h3>
<p>这个脚本的作用很直接。<br>
在 ChatGPT 网页的每一条 AI 回复工具栏里，它会嵌入两个按钮：<strong>「到达最新」</strong> 和 <strong>「返回当前」</strong>。</p>

<h5><a name="p-9516662-h-4" class="anchor" href="https://linux.do#p-9516662-h-4"></a>返回当前：</h5>
<p><img src="https://linux.do/uploads/default/original/4X/7/c/b/7cb297a4e0dca650e48f61d4b9daf0c053dffa2c.png" alt="image" data-base62-sha1="hN7PZY29LMguSZMTn4VItuuJ80k" width="690" height="52" data-dominant-color="FDFDFD"></p>
<h3><a name="p-9516662-h-5" class="anchor" href="https://linux.do#p-9516662-h-5"></a>使用说明：</h3>
<p>当你在旧消息上点击「到达最新」，页面会平滑跳转到当前最新的一条回复，并自动记录你跳转前的位置。到了最新回复后，会出现「返回当前」按钮，让你一键回到刚才的那条旧消息。</p>
<p>新的 AI 回复出现时，脚本会自动完成“角色切换”：<br>
上一条最新回复的按钮变成「到达最新」，新的最新回复获得「返回当前」。<br>
旧消息上的「到达最新」始终指向<strong>当前最新</strong>，不会因为新消息出现而跳错目标。</p>
<h3><a name="p-9516662-h-6" class="anchor" href="https://linux.do#p-9516662-h-6"></a>特点介绍</h3>
<p>按钮与原生工具栏横向排布，位置在“三点菜单”的左边，样式保持一致，不突兀、不挡内容。<br>
没有自动滚动，没有多余动画，纯手动、可控。</p>
<h3><a name="p-9516662-h-7" class="anchor" href="https://linux.do#p-9516662-h-7"></a>使用说明</h3>

// @namespace    kiii
// @version      0.5.0

// @run-at       document-idle
// @grant        none
// ==/UserScript==

  'use strict';
  const $$ = (s, r=document) =&gt; Array.from(r.querySelectorAll(s));
  const SCROLL = { behavior:'smooth', block:'start', inline:'nearest' };
  let lastViewed = null, lastJumpFrom = null, jumping = false;

  const iconLatest = `&lt;svg width="20" height="20" viewBox="0 0 20 20" fill="currentColor"&gt;&lt;path d="M10 3c.37 0 .67.3.67.67v8.72l3.14-3.14a.67.67 0 0 1 .95.95l-4.33 4.33a.67.67 0 0 1-.95 0L5.14 11.2a.67.67 0 1 1 .95-.95l3.25 3.25V3.67c0-.37.3-.67.67-.67Z"/&gt;&lt;/svg&gt;`;
  const iconBack   = `&lt;svg width="20" height="20" viewBox="0 0 20 20" fill="currentColor"&gt;&lt;path d="M7.53 8.47a.75.75 0 0 1 1.06 1.06L7.06 11.06H13a4 4 0 0 1 0 8H6.75a.75.75 0 0 1 0-1.5H13a2.5 2.5 0 0 0 0-5H7.06l1.53 1.53a.75.75 0 1 1-1.06 1.06l-2.75-2.75a.75.75 0 0 1 0-1.06l2.75-2.75Z"/&gt;&lt;/svg&gt;`;

  const throttle = (fn, wait=120) =&gt; { let t=0,h; return (...a)=&gt;{ const n=Date.now(); if(!t||n-t&gt;=wait){t=n;fn(...a);} else { clearTimeout(h); h=setTimeout(()=&gt;{t=Date.now();fn(...a);}, wait-(n-t)); } }; };
  const scrollToEl = el =&gt; { if(!el) return; jumping = true; el.scrollIntoView(SCROLL); setTimeout(()=&gt; jumping=false, 450); };

  const getAllTurns = () =&gt; $$('article[data-turn]');
  const getAssistantTurns = () =&gt; $$('article[data-turn="assistant"]');
  const getLastAssistant = () =&gt; { const a = getAssistantTurns(); return a.at(-1) || null; };

  addEventListener('scroll', throttle(() =&gt; {
    if (jumping) return;
    const turns = getAllTurns(), mid = innerHeight/2;
    let best=null, bestDist=1e9;
    for (const el of turns) {
      const r = el.getBoundingClientRect();
      const d = Math.abs((r.top + r.height/2) - mid);
      if (d &lt; bestDist) { bestDist = d; best = el; }

      const r = el.getBoundingClientRect();
      const d = Math.abs((r.top + r.height/2) - mid);
      if (d &lt; bestDist) { bestDist = d; best = el; }
    }

    const candidates = $$('div', article).filter(d =&gt;
      d.classList.contains('flex') &amp;&amp;
      d.classList.contains('justify-start') &amp;&amp;
      d.querySelector('button[aria-haspopup="menu"]')
    );
    return candidates.at(-1) || null;
  }

    btn.className = 'text-token-text-secondary hover:bg-token-bg-secondary rounded-lg';
    btn.setAttribute('aria-label', label);
    btn.dataset.jumpRole = role;

    btn.addEventListener('click', (e) =&gt; { e.stopPropagation(); onClick?.(); });

  }

  function normalizeToolbar(article, isLatest) {
    const toolbar = findToolbar(article); if (!toolbar) return;
    const kebab = toolbar.querySelector('button[aria-haspopup="menu"]') ||
                  Array.from(toolbar.querySelectorAll('button')).at(-1);

    const kill = sel =&gt; toolbar.querySelectorAll(sel).forEach(b =&gt; b.remove());

    if (isLatest) {
      kill('button[data-jump-role="latest"]');               // 最新不应有“到达最新”
      if (!toolbar.querySelector('button[data-jump-role="back"]')) {
        const bBack = makeButton('返回当前回复', iconBack, () =&gt; {
          const target = (lastJumpFrom &amp;&amp; document.body.contains(lastJumpFrom)) ? lastJumpFrom
                        : (lastViewed &amp;&amp; document.body.contains(lastViewed)) ? lastViewed
                        : getAllTurns().at(0);
          if (!target) return;
          const tmp = lastViewed;
          lastViewed = target;
          lastJumpFrom = tmp || lastJumpFrom;
          scrollToEl(target);
        }, 'back');
        if (kebab &amp;&amp; kebab.parentElement === toolbar) toolbar.insertBefore(bBack, kebab);
        else toolbar.appendChild(bBack);
      }
    } else {
      kill('button[data-jump-role="back"]');                 // 旧消息不应有“返回当前”
      if (!toolbar.querySelector('button[data-jump-role="latest"]')) {
        const bLatest = makeButton('到达最新消息', iconLatest, () =&gt; {
          const latest = getLastAssistant(); if (!latest) return;
          lastJumpFrom = article;
          lastViewed   = latest;
          normalizeToolbar(latest, true); // 确保最新有“返回当前”
          scrollToEl(latest);
        }, 'latest');
        if (kebab &amp;&amp; kebab.parentElement === toolbar) toolbar.insertBefore(bLatest, kebab);
        else toolbar.appendChild(bLatest);
      }
    }
  }

  function normalizeAll() {
    const list = getAssistantTurns();
    const last = list.at(-1);
    list.forEach(a =&gt; normalizeToolbar(a, a === last));
  }

  normalizeAll();
  const mo = new MutationObserver(throttle(() =&gt; normalizeAll(), 120));
  mo.observe(document.documentElement, { childList:true, subtree:true });
})();

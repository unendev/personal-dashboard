            <p>由于l站想查看某人的所有帖子需要点四五步<br>
<strong><a href="https://linux.do/t/topic/1032835">有没有办法快速查看作者的全部帖子 - 搞七捻三 - LINUX DO</a></strong><br>
所以出现了这个让ai帮写的脚本<br>
在 <a href="http://linux.do">linux.do</a> 论坛的每个帖子中，在用户头像下方添加一个“主题”按钮，方便快速查看该用户的所有主题帖。经过源码分析，精准定位。<br>
请大家多多点赞 <img src="https://linux.do/images/emoji/twemoji/smiley.png?v=15" title=":smiley:" class="emoji" alt=":smiley:" loading="lazy" width="20" height="20"></p>
<pre><code class="lang-auto">// ==UserScript==
// @name         Linux.do 用户主题按钮 (最终修正版)
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  在 linux.do 论坛的每个帖子中，在用户头像下方添加一个“主题”按钮，方便快速查看该用户的所有主题帖。经过源码分析，精准定位。
// @author       AI &amp; You
// @match        https://linux.do/t/*
// @grant        none
// @run-at       document-idle
// ==/UserScript==

(function() {
    'use strict';

    // 定义核心函数，用于查找帖子并添加按钮
    const addButtons = () =&gt; {
        // 1. 选择页面中所有独立的帖子容器。
        // `div.topic-post` 是每个帖子的最外层容器。
        const posts = document.querySelectorAll('div.topic-post');

        posts.forEach(post =&gt; {
            // 2. 在每个帖子(post)内部，分别查找头像容器和用户信息元素。
            //    - 头像容器 `.topic-avatar` 是我们添加按钮的目标位置。
            //    - 用户链接 `a[data-user-card]` 含有我们需要的用户名。
            const avatarContainer = post.querySelector('.topic-avatar');
            const userCardLink = post.querySelector('a[data-user-card]');

            // 3. 确保两个元素都找到了，并且这个头像容器里还没有我们添加过的按钮。
            if (avatarContainer &amp;&amp; userCardLink &amp;&amp; !avatarContainer.querySelector('.user-topics-button')) {

                // 4. 从 `data-user-card` 属性中稳定地获取用户名。
                const username = userCardLink.dataset.userCard;

                if (username) {
                    // 构建指向用户主题页面的 URL
                    const topicsUrl = `https://linux.do/u/${username}/activity/topics`;

                    // 创建按钮元素 (一个 `&lt;a&gt;` 标签)
                    const button = document.createElement('a');
                    button.href = topicsUrl;
                    button.textContent = 'TA的主题';
                    button.target = '_blank'; // 设置在新标签页打开
                    button.rel = 'noopener noreferrer'; // 增加安全性
                    button.className = 'user-topics-button'; // 添加一个唯一的类名，用于步骤3的检查

                    // 为按钮添加样式，使其更美观
                    Object.assign(button.style, {
                        display: 'block',
                        textAlign: 'center',
                        marginTop: '8px',
                        padding: '4px 6px',
                        backgroundColor: '#4088da', // 匹配网站风格的蓝色
                        color: 'white',
                        borderRadius: '5px',
                        fontSize: '12px',
                        fontWeight: 'bold',
                        textDecoration: 'none',
                        transition: 'background-color 0.2s',
                    });

                    // 添加鼠标悬停效果
                    button.onmouseover = () =&gt; button.style.backgroundColor = '#2a6cbb';
                    button.onmouseout = () =&gt; button.style.backgroundColor = '#4088da';

                    // 5. 将创建好的按钮添加到头像容器的末尾。
                    avatarContainer.appendChild(button);
                }
            }
        });
    };

    // 页面首次加载时，等待一段时间让内容渲染完再执行
    setTimeout(addButtons, 1000);

    // 论坛是动态加载的 (无限滚动/切换帖子)，所以需要监视页面的变化。
    // 当有新帖子被添加到页面中时，重新运行我们的函数。
    const observer = new MutationObserver((mutations) =&gt; {
        // 为了性能，可以检查一下是否真的有节点添加进来
        let nodesAdded = mutations.some(m =&gt; m.addedNodes &amp;&amp; m.addedNodes.length &gt; 0);
        if (nodesAdded) {
            addButtons();
        }
    });

    // 配置观察器：观察整个文档的变化
    observer.observe(document.body, {
        childList: true,
        subtree: true,
    });

})();


</code></pre>
